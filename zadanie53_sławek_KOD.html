<script type="text/javascript">
    var a_sla = parseInt(prompt("Podaj wysokość prostokąta (a):"));
    var b_sla = parseInt(prompt("Podaj szerokość prostokąta (b):"));

    for (var i = 0; i < b_sla; i++) {
        document.write("*");
    }
    document.write("<br>");

    var wiersz_sla = "*";
    for (var i = 1; i <= (b_sla - 2); i++) {
        wiersz_sla = wiersz_sla + "  ";
    }
    wiersz_sla = wiersz_sla + "*";

    for (var i = 0; i < a_sla - 2; i++) {
        document.write(wiersz_sla + "<br>");
    }

    for (var i = 0; i < b_sla; i++) {
        document.write("*");
    }
</script>
