<script type="text/javascript">
    // Dane osobowe
    var nazwisko = "Slawek";   // bez polskich znaków
    var imie = "Bartosz";

    // Obliczanie a i b
    var a_sła = (nazwisko.length % 4) + 5; // a = (6 % 4) + 5 = 7
    var b_sła = (imie.length % 6) + 20;    // b = (7 % 6) + 20 = 21

    // Górna linia gwiazdek
    for (var i = 0; i < b_sła; i++) {
        document.write("*");
    }
    document.write("<br>");

    // Środkowe linie z pionowymi gwiazdkami
    for (var w = 0; w < a_sła - 2; w++) {
        document.write("*");
        for (var j = 0; j < b_sła - 2; j++) {
            document.write(" ");
        }
        document.write("*<br>");
    }

    // Dolna linia gwiazdek
    for (var i = 0; i < b_sła; i++) {
        document.write("*");
    }

    // Wyświetlenie wartości a i b
    document.write("<br><br>");
    document.write("a = " + a_sła + "<br>");
    document.write("b = " + b_sła + "<br>");
</script>
