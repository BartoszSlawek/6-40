    <script type="text/javascript">
        // Program do obliczania silni liczby podanej przez użytkownika
        // Zmienna n_sła zawiera liczbę podaną przez użytkownika
        let n_sła = parseInt(prompt("Podaj liczbę, dla której chcesz obliczyć silnię:"));

        // Zmienna silnia_sła będzie przechowywać wynik końcowy
        let silnia_sła;

        // Jeżeli liczba to 0, to zgodnie z definicją 0! = 1
        if (n_sła === 0) {
            silnia_sła = 1;
        } else {
            // Inicjalizacja wyniku na 1
            silnia_sła = 1;

            // Pętla mnożąca kolejne liczby od 1 do n_sła
            for (let i = 1; i <= n_sła; i++) {
                silnia_sła *= i; // Mnożenie dotychczasowego wyniku przez i
            }
        }

        // Wyświetlenie wyniku na stronie
        document.write(n_sła + "! = " + silnia_sła);
    </script>
