            var h_sła = 0;
            do {
                var h_sła = prompt("Podaj wysokość trójkąta (większą od zera):", "");
                h_sła = parseFloat(h_sła);
            } while (h_sła <= 0);

            var a_sła = 0;
            do {
                var a_sła = prompt("Podaj podstawę trójkąta (większą od zera):", "");
                a_sła = parseFloat(a_sła);
            } while (!(a_sła > 0));

            var Pole = (a_sła * h_sła) / 2;
            document.write("Pole=" + Pole.toFixed(2) + " cm^2" + "<br>");
        </script>
